    <script>
        // ============ CLOCK & DATE ============
        function updateClock() {
            const now = new Date();
            const hours = String(now.getHours()).padStart(2, '0');
            const minutes = String(now.getMinutes()).padStart(2, '0');
            const seconds = String(now.getSeconds()).padStart(2, '0');
            document.getElementById('clock').textContent = `${hours}:${minutes}`;
            
            const options = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
            document.getElementById('date').textContent = now.toLocaleDateString('en-US', options);
            
            const h = now.getHours();
            const greet = h < 12 ? 'Morning' : h < 18 ? 'Afternoon' : 'Evening';
            document.getElementById('greeting').textContent = greet;
        }
        updateClock();
        setInterval(updateClock, 1000);

        // ============ HARDWARE STATS (Simulated) ============
        let cpuVal = 25, ramVal = 45, gpuVal = 15, downVal = 2.5, upVal = 0.8;
        
        function updateHardware() {
            cpuVal = Math.min(95, Math.max(8, cpuVal + (Math.random() - 0.5) * 20));
            ramVal = Math.min(90, Math.max(30, ramVal + (Math.random() - 0.5) * 8));
            gpuVal = Math.min(85, Math.max(5, gpuVal + (Math.random() - 0.5) * 15));
            downVal = Math.max(0.1, downVal + (Math.random() - 0.5) * 3);
            upVal = Math.max(0.05, upVal + (Math.random() - 0.5) * 1);
            
            const cpuTemp = Math.round(38 + cpuVal * 0.5);
            const gpuTemp = Math.round(40 + gpuVal * 0.4);
            const ping = Math.round(15 + Math.random() * 20);
            
            document.getElementById('cpuPercent').textContent = Math.round(cpuVal) + '%';
            document.getElementById('cpuBar').style.width = cpuVal + '%';
            document.getElementById('cpuTemp').textContent = cpuTemp + '°C';
            document.getElementById('cpuTemp').style.color = cpuTemp > 70 ? '#ef4444' : cpuTemp > 55 ? '#f59e0b' : '#9ca3af';
            
            const ramUsed = (ramVal / 100 * 16).toFixed(1);
            document.getElementById('ramPercent').textContent = Math.round(ramVal) + '%';
            document.getElementById('ramBar').style.width = ramVal + '%';
            document.getElementById('ramUsed').textContent = ramUsed + ' / 16 GB';
            
            document.getElementById('gpuPercent').textContent = Math.round(gpuVal) + '%';
            document.getElementById('gpuBar').style.width = gpuVal + '%';
            document.getElementById('gpuTemp').textContent = gpuTemp + '°C';
            document.getElementById('gpuTemp').style.color = gpuTemp > 70 ? '#ef4444' : gpuTemp > 55 ? '#f59e0b' : '#9ca3af';
            
            document.getElementById('netDown').textContent = downVal.toFixed(1) + ' MB/s';
            document.getElementById('netUp').textContent = upVal.toFixed(1) + ' MB/s';
            document.getElementById('ping').textContent = ping + 'ms';
        }
        updateHardware();
        setInterval(updateHardware, 2000);

        // ============ STORAGE DRIVES ============
        const drives = [
            { label: 'C: Drive', name: 'Windows', used: 320, total: 500, color: 'blue' },
            { label: 'D: Drive', name: 'Data', used: 850, total: 2000, color: 'purple' },
            { label: 'E: Drive', name: 'Games', used: 1200, total: 2000, color: 'green' },
            { label: 'F: Drive', name: 'Media', used: 3800, total: 4000, color: 'yellow' },
            { label: 'External', name: 'Backup', used: 150, total: 1000, color: 'orange' },
            { label: 'NVMe', name: 'Scratch', used: 450, total: 1000, color: 'red' }
        ];

        function renderDrives() {
            const grid = document.getElementById('driveGrid');
            grid.innerHTML = drives.map(d => {
                const pct = Math.round(d.used / d.total * 100);
                const colorMap = { blue: 'from-blue-500 to-blue-400', purple: 'from-purple-500 to-purple-400', green: 'from-green-500 to-green-400', yellow: 'from-yellow-500 to-yellow-400', orange: 'from-orange-500 to-orange-400', red: 'from-red-500 to-red-400' };
                return `
                    <div class="glass-light rounded-xl p-4">
                        <div class="flex items-center justify-between mb-2">
                            <span class="text-sm font-medium">${d.label} <span class="text-gray-500">(${d.name})</span></span>
                            <span class="text-xs ${pct > 90 ? 'text-red-400' : 'text-gray-400'}">${pct}%</span>
                        </div>
                        <div class="w-full h-2.5 bg-white/5 rounded-full overflow-hidden">
                            <div class="progress-bar h-full bg-gradient-to-r ${colorMap[d.color]} rounded-full" style="width: ${pct}%"></div>

        let downloadQueue = [];
        let downloadInterval = null;

        function startDownload() {
            const url = document.getElementById('videoUrl').value.trim();
            if (!url) return;
            
            const format = document.getElementById('formatSelect').value;
            const folderSelect = document.getElementById('folderSelect').value;
            const folder = folderSelect === 'custom' ? document.getElementById('customFolder').value : folderSelect;
            
            const id = Date.now();
            const item = { id, url, format, folder, progress: 0, status: 'downloading', speed: 0, eta: '--' };
            downloadQueue.unshift(item);
            document.getElementById('videoUrl').value = '';
            
            renderQueue();
            simulateDownload(id);
        }

        function addPlaylist() {
            const url = document.getElementById('videoUrl').value.trim();
            if (!url) return;
            const format = document.getElementById('formatSelect').value;
            const folderSelect = document.getElementById('folderSelect').value;
            const folder = folderSelect === 'custom' ? document.getElementById('customFolder').value : folderSelect;
            
            for (let i = 1; i <= 5; i++) {
                const id = Date.now() + i;
                downloadQueue.push({ id, url: url + `&index=${i}`, format, folder, progress: 0, status: 'queued', speed: 0, eta: '--' });
            }
            document.getElementById('videoUrl').value = '';
            renderQueue();
            processQueue();
        }

        function simulateDownload(id) {
            const item = downloadQueue.find(q => q.id === id);
            if (!item) return;
            
            item.status = 'downloading';
            renderQueue();
            
            const iv = setInterval(() => {
                const qItem = downloadQueue.find(q => q.id === id);
                if (!qItem || qItem.progress >= 100) {
                    clearInterval(iv);
                    if (qItem) qItem.status = 'completed';
                    renderQueue();
                    return;
                }
                qItem.progress += Math.random() * 8 + 2;
                qItem.speed = (Math.random() * 15 + 5).toFixed(1);
                qItem.eta = Math.round((100 - qItem.progress) / 8) + 's';
                if (qItem.progress > 100) qItem.progress = 100;
                renderQueue();
            }, 500);
        }

        function processQueue() {
            const queued = downloadQueue.filter(q => q.status === 'queued');
            if (queued.length > 0) {
                setTimeout(() => {
                    simulateDownload(queued[0].id);
                    if (queued.length > 1) processQueue();
                }, 2000);
            }
        }

        function renderQueue() {
            const container = document.getElementById('downloadQueue');
            container.classList.toggle('hidden', downloadQueue.length === 0);
            container.innerHTML = downloadQueue.map(q => {
                const statusColor = q.status === 'completed' ? 'text-green-400' : q.status === 'downloading' ? 'text-blue-400' : 'text-gray-500';
                const statusIcon = q.status === 'completed' ? 'fa-check-circle' : q.status === 'downloading' ? 'fa-spinner fa-spin' : 'fa-clock-o';
                return `
                    <div class="glass-light rounded-xl p-3">
                        <div class="flex items-center justify-between mb-2">
                            <span class="text-xs text-gray-400 truncate flex-1 mr-2">${q.url}</span>
                            <span class="${statusColor} text-xs"><i class="fa ${statusIcon} mr-1"></i>${q.status === 'completed' ? 'Done' : q.status === 'downloading' ? q.progress.toFixed(0) + '%' : 'Queued'}</span>
                        </div>
                        <div class="w-full h-1.5 bg-white/5 rounded-full overflow-hidden">
                            <div class="h-full ${q.status === 'completed' ? 'bg-green-500' : 'bg-blue-500'} rounded-full transition-all duration-300" style="width: ${q.progress}%"></div>
                        </div>
                        ${q.status === 'downloading' ? `<div class="flex justify-between mt-1"><span class="text-xs text-gray-500">${q.speed} MB/s</span><span class="text-xs text-gray-500">ETA: ${q.eta}</span></div>` : ''}
                    </div>
                `;
            }).join('');
        }

        let todos = JSON.parse(localStorage.getItem('dashTodos')) || [];

        function saveTodos() {
            localStorage.setItem('dashTodos', JSON.stringify(todos));
        }

        function addTodo() {
            const input = document.getElementById('todoInput');
            const text = input.value.trim();
            if (!text) return;
            const priority = document.getElementById('todoPriority').value;
            todos.unshift({ id: Date.now(), text, completed: false, priority, created: new Date().toISOString() });
            input.value = '';
            saveTodos();
            renderTodos();
        }

        function toggleTodo(id) {
            const todo = todos.find(t => t.id === id);
            if (todo) todo.completed = !todo.completed;
            saveTodos();
            renderTodos();
        }

        function deleteTodo(id) {
            todos = todos.filter(t => t.id !== id);
            saveTodos();
            renderTodos();
        }

        function editTodo(id) {
            const todo = todos.find(t => t.id === id);
            if (!todo) return;
            const newText = prompt('Edit task:', todo.text);
            if (newText !== null && newText.trim()) {
                todo.text = newText.trim();
                saveTodos();
                renderTodos();
            }
        }

        function clearCompleted() {
            todos = todos.filter(t => !t.completed);
            saveTodos();
            renderTodos();
        }

        function renderTodos() {
            const filter = document.getElementById('todoFilter').value;
            let filtered = todos;
            if (filter === 'active') filtered = todos.filter(t => !t.completed);
            if (filter === 'completed') filtered = todos.filter(t => t.completed);
            
            const remaining = todos.filter(t => !t.completed).length;
            document.getElementById('todoCount').textContent = `${remaining} task${remaining !== 1 ? 's' : ''} remaining`;
            
            const container = document.getElementById('todoList');
            const empty = document.getElementById('emptyTodo');
            
            if (filtered.length === 0) {
                container.innerHTML = '';
                empty.classList.remove('hidden');
                return;
            }
            
            empty.classList.add('hidden');
            
            const priorityColors = { low: 'bg-green-500/20 text-green-400 border-green-500/30', medium: 'bg-yellow-500/20 text-yellow-400 border-yellow-500/30', high: 'bg-red-500/20 text-red-400 border-red-500/30' };
            
            container.innerHTML = filtered.map(t => `
                <div class="todo-item ${t.completed ? 'completed' : ''} glass-light rounded-xl px-4 py-3 flex items-center gap-3 group">
                    <button onclick="toggleTodo(${t.id})" class="w-5 h-5 rounded-full border-2 ${t.completed ? 'bg-teal-500 border-teal-500' : 'border-gray-500 hover:border-teal-400'} flex items-center justify-center transition flex-shrink-0">
                        ${t.completed ? '<i class="fa fa-check text-white text-xs"></i>' : ''}
                    </button>
                    <span class="todo-text text-sm flex-1 ${t.completed ? 'line-through text-gray-500' : 'text-gray-200'}">${escapeHtml(t.text)}</span>
                    <span class="text-xs px-2 py-0.5 rounded-full border ${priorityColors[t.priority]}">${t.priority}</span>
                    <div class="flex gap-1 opacity-0 group-hover:opacity-100 transition">
                        <button onclick="editTodo(${t.id})" class="w-6 h-6 rounded-lg hover:bg-white/10 flex items-center justify-center text-gray-400 transition">
                            <i class="fa fa-pencil text-xs"></i>
                        </button>
                        <button onclick="deleteTodo(${t.id})" class="w-6 h-6 rounded-lg hover:bg-red-500/20 flex items-center justify-center text-gray-400 hover:text-red-400 transition">
                            <i class="fa fa-trash text-xs"></i>
                        </button>
                    </div>
                </div>
            `).join('');
        }

        function escapeHtml(text) {
            const div = document.createElement('div');
            div.textContent = text;
            return div.innerHTML;
        }
